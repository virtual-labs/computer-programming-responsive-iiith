void quicksort(int low, int high)
{
int pivot, i, j, temp, size;
int A[size];
if (low < high)
{
pivot = low;
i = low;
j = high;
while (i < j)
{
while (A[i] <= A[pivot] && i <= high)
{
i++;
}
while (A[j] > A[pivot] && j >= low)
{
j--;
}
if (i < j)
{
temp = A[i];
A[i] = A[j];
A[j] = temp;
}
}
temp = A[j];
A[j] = A[pivot];
A[pivot] = temp;
quicksort(low, j - 1);
quicksort(j + 1, high);
}
